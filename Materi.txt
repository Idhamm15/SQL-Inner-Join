CREATE TABLE mahasiswa (
    nim CHAR(8),
    nama VARCHAR(35),
    jk CHAR(9),
    email VARCHAR(35),
    no_hp VARCHAR(14),
    agama VARCHAR(10),
    alamat VARCHAR(300),
    tempat_lahir VARCHAR(20),
    tanggal_lahir DATE
);
